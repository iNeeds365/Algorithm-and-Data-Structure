\begin{function}
	{\bf Edge classification}\\
	\underline{{\bf Function} DFS} $()$\\
	\ForEach{$v$ {\bf in} $V$}{
		visited[$v$] = $false;$\\
		pre[$v$] = $null$;\\
		post[$v$] = $null$;\\
	}
	
	\ForEach{$v$ {\bf in} $V$}{
		\If{visited[$v$]==$false$}
		{
			explore($v$);\\
		}
	}

	\hbox{}
	\underline{{\bf Function} explore} $(z)$\;
	pre[$z$] = clock ++; \\
	visited[$z$] = true; \\
	\ForEach{$(z,w)$ {\bf in} $E$}{
		\If{visited[$w$]==false}{
			print: (z,w) is a tree edge\\
			explore($w$);\\
		}
		\ElseIf{pre[z] $<$ pre[w]}
		{
			print: (z,w) is a forward edge
		}
		\ElseIf{post[w] == null}
		{
			print: (z,w) is a back edge
		}
		\Else
		{
			print: (z,w) is a cross edge
		}
	}
	post[$z$] = clock++;
\end{function}
